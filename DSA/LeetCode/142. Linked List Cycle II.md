# [142. Linked List Cycle II](https://leetcode.com/problems/linked-list-cycle-ii/)

```c++
class Solution {
public:
    ListNode *detectCycle(ListNode *head) {
        if(head == nullptr){
            return head;
        }
         
        ListNode* fast = head;
        ListNode* slow = head;
        while(fast != nullptr && fast->next !=nullptr){
            fast=fast->next->next;
            slow=slow->next;

            if(fast == slow) {
                // pointer reached meeting point
                // which means cycle is present
                slow=head;
                while(fast != slow){
                    fast=fast->next;
                    slow=slow->next;
                }
                return slow;
            }
        }
        // cycle is absent
        return nullptr;
    }
};
```
